	    \section*{Single Operation structures (Group-Like)}
	    \mbox{}\\
		Consider a set $M$ and a binary operation:
		\begin{displaymath}
			\OpA : M \times M \rightarrow M
		\end{displaymath}
		
		Consider $X \subset M$.	
		\vfill
		\begin{definition}[Subgroup]
			$X$ is closed w.r.t. $\OpA$, i.e.:
				\begin{equation}\label{def:SubGroup}
					\ran(\OpA \vert_X )\subset X \quad \textrm{i.e.} \quad	( x \OpA y ) \in X \qquad \forall x,y\in C
				\end{equation}	
		\end{definition}
		\vfill
		%
		\begin{definition}[Ideal]
			$X$ is a subgroup s.t.:
				\begin{equation}\label{def:Ideal}
					X \OpA M \subset X \quad \textrm{i.e.} \quad	( x \OpA y ) \in X \qquad \forall x\in X, y \in M
				\end{equation}		
		
		\end{definition}
		\vfill
	\end{minipage}
	%
	\hspace{1cm}
	%

    		\hline
    	\end{tabular}
    	\end{center}
		%
		\begin{definition}[Total ~\emph{(totality)}]
			\begin{equation}\label{Totality}
				\dom(\OpA) = M \times M
			\end{equation}
		\end{definition}
		\vfill
		%
		\begin{definition}[Associative ~\emph{(associativity)}]
			\begin{equation}\label{Associativity}
				(x \OpA y) \OpA z = x \OpA (y \OpA z) \qquad \forall x,y,z \in M
			\end{equation}	
		\end{definition}
		%
		\vfill
		\begin{definition}[Unital ~\emph{(identity)}]
			\begin{equation}\label{Identity}
				\exists e \in M \quad : \quad e \OpA x = x \OpA e = x \quad \forall x \in M
			\end{equation}
			Such an element is unique, and thus one speaks of \emph{neutral element}.	
		\end{definition}
		%
		\vfill
		\begin{definition}[Invertible ~\emph{(invertibility)}]
			\emph{Exists Inverse:}
				\begin{equation}\label{Invertibility}
					\forall a \in M \quad \exists b \in M \quad : \quad a \OpA b = b \OpA a = e
				\end{equation}
			where e is the neutral element and thus one speaks of \emph{identity element} .
		\end{definition}
		%
		\vfill
		\begin{definition}[Commutative ~\emph{(commutativity)}]
			\begin{equation}\label{Commutativity}
				x \OpA y = y \OpA x \qquad \forall x,y\in M
			\end{equation}
		\end{definition}
		\vfill

	    \section*{Double Operation Structures (Ring-Like)}
	    \mbox{}\\
		%
		Consider a set $R$ and two binary operation as above:
		\begin{displaymath}
				\OpA : R \times R \rightarrow R \qquad 	\OpB : R \times R \rightarrow R
		\end{displaymath}
		
		Are defined some basic compatibility relations:
		%
		\vfill
		\begin{definition}[$\OpA$ is \emph{Left distributivite} with respect to $\OpB$]
			\begin{equation}\label{LeftDistributivity}
				a \OpA ( b \OpB c) = (a \OpA b) \OpB (a \OpA c) \qquad \forall a,b,c \in R
			\end{equation}		
		\end{definition}
		%
		\vfill
		\begin{definition}[$\OpA$ is \emph{Right distributivite} with respect to $\OpB$]
			\begin{equation}\label{RightDistributivity}
				( a \OpB b)\OpA c = (a \OpA c) \OpB (b \OpA c) \qquad \forall a,b,c \in R	
			\end{equation}
		\end{definition}
		%			
		

		\begin{notation}
			Mimicking the real number field we denote
			\begin{displaymath}
			\begin{tabular}{|c|c c|}
				\hline 
					& Neutral & Inverse \\
				\hline
				$\OpA$ & $\mathbb{1}$ & $(\cdot)^{-1}$\\
				$\OpB$ & $\mathbb{O}$ & $-(\cdot)$\\
				\hline
			\end{tabular}			
			\end{displaymath}

		\end{notation}
		
		Basic properties following from the axioms:
		\begin{itemize}
			\item The additive identity, the additive inverse of each element, and the multiplicative identity are unique.
			\item For any element $x$ in a ring $R$, one has $x 0 = 0 = 0x $ (zero is an \emph{absorbing element} with respect to multiplication) and $(-1)x = -x$.\\
			(follows that invertibility w.r.t. multiplication can be imposed only on $a \neq =0$)
			\item If $0 = 1$ in a ring $R$ (or more generally, 0 is a unit element), then $R$ has only one element, and is called the \emph{zero ring}.
			\item Distributivity relations could be regarded as linear action of $(R, \OpB)$ on $(R, \OpA)$.
		\end{itemize}

	    \section*{Abelian groups with a Ring action (Module-like)}
	    \mbox{}\\
	    		Let $(A,\OpB,e)$ be an abelian group. \\
	    		Consider a ring $(\KField,+,\cdot, 1)$.
	    		
	    		\begin{definition}[Scalar multiplication]
	    			Is a ring action
	    			\begin{displaymath}
	    				\morphism{\ast}
	    				{\KField\times A}{A}
	    				{(\lambda,a)}{\lambda a}
	    			\end{displaymath}
	    			satisfying compatibility relations ($\forall \lambda,\delta \in \KField$ and $\forall \mathbf{a},\mathbf{b} \in A$):
		%:

			\begin{itemize}
			\item \emph{(Ring action)} :
				\begin{equation}\label{scalarmulti}
					\lambda( \delta  \mathbf{a}) = (\lambda\cdot \delta)  \mathbf{a}
				\end{equation}
			\item \emph{Compatibility of scalar multiplication} w.r.t. $\OpB$:
				\begin{equation}\label{scalarcompat}
					 (\lambda \mathbf{a} ) \OpB (\delta \mathbf{b)}= (\lambda\cdot \delta) (\mathbf{a}\OpB \mathbf{b})
				\end{equation}
			\item \emph{Identity element of scalar multiplication}:
				\begin{equation}\label{scalaridentity}
					1 \mathbf{a} = e \OpB \mathbf{a}  
				\end{equation}
			\item \emph{Distributivity of scalar multiplication} w.r.t $\OpB$:
				\begin{equation}\label{scalarDistributivityA}
					\lambda ( \mathbf{a} \OpB \mathbf{b} ) = \lambda \mathbf{a} \OpB a \mathbf{b} 
				\end{equation}
			\item \emph{Distributivity of scalar multiplication with respect to field addition}:
				\begin{equation}\label{scalarDistributivityB}
					(\lambda + \delta ) \mathbf{a} = \lambda  \mathbf{a}  \OpB  \delta \mathbf{a}
				\end{equation}
			\end{itemize}

	    			
	    		\end{definition}

	    		Consider a ring $(\KField,+,\cdot, 1)$.
	    		Assume $(A,\OpB, \KField)$ to be a $\KField$-module with action $\cdot$ and $(A,\OpA)$ being a magma.
	    		
	    		\begin{definition}[Algebra compatibility of $\cdot$ and $\OpA$]
	    			Scalar multiplication $\cdot$ and magma product $\OpA$ satisfy the following compatibility relation
	    			\begin{align*}
	    			\lambda \cdot (\mathbf{a}\OpA \mathbf{b}) 
	    				&=~
	    				(\lambda \mathbf{a})\OpA \mathbf{b}
	    				\\
	    				&=~
	    				\mathbf{b}	  \OpA   		(\lambda \mathbf{b})		
	    			\end{align*}
	    		
	    		\end{definition}
